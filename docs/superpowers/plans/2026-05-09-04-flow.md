```python
"""项目内共享的数据类"""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any


@dataclass
class StartableFlow:
    """可发起的流程"""
    flow_id: str
    flow_code: str | None
    flow_name: str
    category_name: str | None
    starter_username: str
    raw: dict[str, Any] = field(default_factory=dict)


@dataclass
class FlowNodeDescriptor:
    """流程节点描述"""
    node_id: str
    node_name: str
    node_type: str
    branch_id: str | None
    operation_types: list[str]
    requires_assignment: bool
    assignment_targets: list[AssignmentTarget]
    action_configs: list[NodeActionConfig]
    raw: dict[str, Any] = field(default_factory=dict)


@dataclass
class AssignmentTarget:
    """节点指派目标"""
    field: str
    selector_type: str  # user / department / position / role / expression
    required: bool
    candidate_source: str | None


@dataclass
class NodeActionConfig:
    """节点动作配置"""
    action_type: str   # approve / reject / countersign / transfer / revoke
    handler_source: str # assigned_user / candidate_pool / role / department / expression / todo_owner
    requires_login_user: bool
    candidate_source: str | None
    raw: dict[str, Any] = field(default_factory=dict)


@dataclass
class BranchDescriptor:
    """分支描述"""
    branch_id: str
    branch_name: str
    parent_branch_id: str | None
    condition: str | None  # 条件分支的触发条件
    raw: dict[str, Any] = field(default_factory=dict)


@dataclass
class ParsedFlow:
    """解析后的完整流程结构"""
    flow_id: str
    flow_name: str
    start_form_fields: list[FormField]
    nodes: list[FlowNodeDescriptor]
    branches: list[BranchDescriptor]
    raw: dict[str, Any] = field(default_factory=dict)


@dataclass
class FormField:
    """表单字段"""
    field_id: str
    field_name: str
    field_type: str  # text / number / date / select / textarea / ...
    required: bool
    readonly: bool
    options: list[str] | None  # 下拉选项
    default_value: Any | None
    raw: dict[str, Any] = field(default_factory=dict)


@dataclass
class UserBrief:
    """用户简要信息"""
    user_id: str
    username: str
    display_name: str
    department_id: str | None
    department_name: str | None
    position_id: str | None
    position_name: str | None
    enabled: bool


@dataclass
class DepartmentBrief:
    """部门简要信息"""
    department_id: str
    department_name: str
    parent_id: str | None


@dataclass
class PositionBrief:
    """岗位简要信息"""
    position_id: str
    position_name: str
    department_id: str | None
```

```python
"""HTTP API 客户端封装（httpx.AsyncClient）"""
from __future__ import annotations

import httpx


class ApiClient:
    """统一的异步 HTTP 客户端"""

    def __init__(self, base_url: str, timeout: float = 30.0):
        self._base_url = base_url.rstrip("/")
        self._timeout = timeout
        self._client: httpx.AsyncClient | None = None

    async def __aenter__(self) -> ApiClient:
        self._client = httpx.AsyncClient(
            base_url=self._base_url,
            timeout=httpx.Timeout(self._timeout),
            follow_redirects=True,
        )
        return self

    async def __aexit__(self, *args) -> None:
        if self._client:
            await self._client.aclose()

    def set_sid(self, sid: str) -> None:
        """注入 SID 到请求头"""
        if self._client:
            self._client.headers["X-SID"] = sid

    async def get(
        self,
        path: str,
        params: dict | None = None,
        headers: dict | None = None,
    ) -> httpx.Response:
        if not self._client:
            raise RuntimeError("ApiClient not entered")
        merged_headers = dict(self._client.headers)
        if headers:
            merged_headers.update(headers)
        return await self._client.get(path, params=params, headers=merged_headers)

    async def post(
        self,
        path: str,
        json: dict | None = None,
        data: dict | None = None,
        headers: dict | None = None,
    ) -> httpx.Response:
        if not self._client:
            raise RuntimeError("ApiClient not entered")
        merged_headers = dict(self._client.headers)
        if headers:
            merged_headers.update(headers)
        return await self._client.post(path, json=json, data=data, headers=merged_headers)

    async def put(
        self,
        path: str,
        json: dict | None = None,
        headers: dict | None = None,
    ) -> httpx.Response:
        if not self._client:
            raise RuntimeError("ApiClient not entered")
        merged_headers = dict(self._client.headers)
        if headers:
            merged_headers.update(headers)
        return await self._client.put(path, json=json, headers=merged_headers)

    async def delete(
        self,
        path: str,
        headers: dict | None = None,
    ) -> httpx.Response:
        if not self._client:
            raise RuntimeError("ApiClient not entered")
        merged_headers = dict(self._client.headers)
        if headers:
            merged_headers.update(headers)
        return await self._client.delete(path, headers=merged_headers)
```

```python
"""FlowCatalogService：按发起账号查询可发起流程"""
from __future__ import annotations

from workflow_test_desktop.core.session.manager import SessionManager
from workflow_test_desktop.core.config.environment import AppConfig
from workflow_test_desktop.data.models import StartableFlow


class FlowCatalogService:
    """
    用发起人 SID 查询可发起流程列表。
    屏蔽后端 API 差异，返回统一的 StartableFlow 列表。
    """

    def __init__(self, config: AppConfig, http_client, session_manager: SessionManager):
        self._config = config
        self._http = http_client
        self._sm = session_manager

    async def list_startable_flows(
        self,
        env_id: str,
        username: str,
        search: str | None = None,
        category: str | None = None,
    ) -> list[StartableFlow]:
        """获取指定账号可发起的流程列表"""
        # 获取发起人会话
        lease = await self._sm.get_session(env_id, "user_login", username)
        self._http.set_sid(lease.sid)

        # 构建查询参数
        params = {}
        if search:
            params["keyword"] = search
        if category:
            params["category"] = category

        # 调用后端
        resp = await self._http.get(self._config.flow_catalog_path, params=params)
        resp.raise_for_status()
        raw_list = resp.json().get("data", []) if isinstance(resp.json(), dict) else resp.json()

        # 映射为 StartableFlow
        flows = []
        for item in raw_list:
            flows.append(StartableFlow(
                flow_id=item["flowId"] or item["id"],
                flow_code=item.get("flowCode"),
                flow_name=item["flowName"],
                category_name=item.get("categoryName"),
                starter_username=username,
                raw=item,
            ))
        return flows
```

### Step 5: 创建 `workflow_test_desktop/core/flow/inspector.py`（FlowInspector）

```python
"""FlowInspector：解析流程定义，返回节点+分支+表单"""
from __future__ import annotations

from workflow_test_desktop.core.session.manager import SessionManager
from workflow_test_desktop.core.config.environment import AppConfig
from workflow_test_desktop.data.models import (
    FlowNodeDescriptor, AssignmentTarget, NodeActionConfig,
    BranchDescriptor, ParsedFlow, FormField,
)


class FlowInspector:
    """
    解析流程定义，返回：
    - 起始表单字段
    - 节点列表（含审核配置）
    - 分支关系
    不在解析阶段假定节点只能是"选人"或"选部门"。
    """

    def __init__(self, config: AppConfig, http_client, session_manager: SessionManager):
        self._config = config
        self._http = http_client
        self._sm = session_manager

    async def inspect(
        self,
        env_id: str,
        username: str,
        flow_id: str,
    ) -> ParsedFlow:
        """解析指定流程，返回完整结构"""
        # 获取会话
        lease = await self._sm.get_session(env_id, "user_login", username)
        self._http.set_sid(lease.sid)

        # 获取流程详情
        resp = await self._http.get(
            self._config.flow_detail_path,
            params={"flowId": flow_id},
        )
        resp.raise_for_status()
        raw = resp.json()
        data = raw.get("data", raw) if isinstance(raw, dict) else raw

        # 解析表单字段
        form_fields = self._parse_form_fields(data.get("formConfig", []))

        # 解析分支
        branches = self._parse_branches(data.get("branches", []))

        # 解析节点
        nodes = self._parse_nodes(data.get("nodes", []), branches)

        return ParsedFlow(
            flow_id=flow_id,
            flow_name=data.get("flowName", ""),
            start_form_fields=form_fields,
            nodes=nodes,
            branches=branches,
            raw=data,
        )

    def _parse_form_fields(self, form_config: list) -> list[FormField]:
        """解析表单字段"""
        fields = []
        for item in form_config:
            fields.append(FormField(
                field_id=item["fieldId"],
                field_name=item.get("label", item["fieldId"]),
                field_type=item.get("type", "text"),
                required=item.get("required", False),
                readonly=item.get("readonly", False),
                options=item.get("options"),
                default_value=item.get("default"),
                raw=item,
            ))
        return fields

    def _parse_branches(self, branches_config: list) -> list[BranchDescriptor]:
        """解析分支"""
        branches = []
        for item in branches_config:
            branches.append(BranchDescriptor(
                branch_id=item["branchId"],
                branch_name=item.get("branchName", item["branchId"]),
                parent_branch_id=item.get("parentBranchId"),
                condition=item.get("condition"),
                raw=item,
            ))
        return branches

    def _parse_nodes(self, nodes_config: list, branches: list[BranchDescriptor]) -> list[FlowNodeDescriptor]:
        """解析节点（含审核配置）"""
        descriptors = []
        for item in nodes_config:
            # 解析指派目标
            assignment_targets = []
            for target in item.get("assignmentTargets", []):
                assignment_targets.append(AssignmentTarget(
                    field=target["field"],
                    selector_type=target.get("selectorType", "user"),
                    required=target.get("required", False),
                    candidate_source=target.get("candidateSource"),
                ))

            # 解析动作配置
            action_configs = []
            for action in item.get("actionConfigs", []):
                action_configs.append(NodeActionConfig(
                    action_type=action["actionType"],
                    handler_source=action.get("handlerSource", "assigned_user"),
                    requires_login_user=action.get("requiresLoginUser", True),
                    candidate_source=action.get("candidateSource"),
                    raw=action,
                ))

            descriptors.append(FlowNodeDescriptor(
                node_id=item["nodeId"],
                node_name=item.get("nodeName", item["nodeId"]),
                node_type=item.get("nodeType", "approval"),
                branch_id=item.get("branchId"),
                operation_types=item.get("operationTypes", ["approve", "reject"]),
                requires_assignment=bool(assignment_targets),
                assignment_targets=assignment_targets,
                action_configs=action_configs,
                raw=item,
            ))
        return descriptors
```
