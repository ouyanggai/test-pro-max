def build_summary_item(label: str, value: str) -> str:
    return (
        f'<div class="summary-item">'
        f'<div class="summary-label">{label}</div>'
        f'<div class="summary-value">{value}</div>'
        f'</div>'
    )


def build_node_item(node_name: str, status: str, duration_ms: int) -> str:
    icon = {"completed": "&#10003;", "failed": "&#10007;", "running": "&#9679;", "pending": "&#9675;"}.get(status, "&#9675;")
    return (
        f'<div class="node-item {status}">'
        f'{icon} {node_name} <span>{duration_ms}ms</span>'
        f'</div>'
    )


def build_log_entry(timestamp: str, username: str, action: str, reason: str) -> str:
    return (
        f'<div class="log-entry">'
        f'[{timestamp}] <span class="action">{action}</span> '
        f'{username} &#8212; {reason}'
        f'</div>'
    )

### Step 2: 创建 `workflow_test_desktop/core/export/exporter.py`（ReportExporter）

```python
"""ReportExporter：读取 SQLite 生成 HTML / JSON 报告"""
from __future__ import annotations

import json
import time
from pathlib import Path
from typing import Any

from workflow_test_desktop.core.storage.db import Database
from workflow_test_desktop.core.export.templates import (
    HTML_TEMPLATE, build_summary_item, build_node_item, build_log_entry,
)


class ReportExporter:
    """
    报告导出器。
    敏感字段已在 RunRecorder 层统一脱敏，此处直接使用。
    """

    def __init__(self, db: Database):
        self._db = db

    async def export_html(self, execution_id: str, output_path: str | Path) -> Path:
        """生成 HTML 报告"""
        data = await self._load_report_data(execution_id)
        html = HTML_TEMPLATE.format(
            title=f"执行报告 - {data['flow_name']} - {time.strftime('%Y-%m-%d %H:%M:%S')}",
            bg_primary="#FFFFFF", bg_card="#FFFFFF", border="#E2E4E9",
            text_primary="#1A1D26", text_secondary="#6B7280",
            accent="#0EA5E9", success="#10B981", warning="#F59E0B", error="#EF4444",
            summary_items=self._build_summary(data),
            branch_details=self._build_branch_details(data),
            session_log=self._build_session_log(data),
        )
        path = Path(output_path)
        path.write_text(html, encoding="utf-8")
        return path

    async def export_json(self, execution_id: str, output_path: str | Path) -> Path:
        """生成 JSON 报告"""
        data = await self._load_report_data(execution_id)
        path = Path(output_path)
        path.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
        return path

    async def _load_report_data(self, execution_id: str) -> dict[str, Any]:
        """从数据库加载报告所需数据"""
        async with self._db.cursor() as cur:
            await cur.execute("SELECT * FROM execution_run WHERE id=?", (execution_id,))
            execution = await cur.fetchone()
            if not execution:
                return {}
            await cur.execute("SELECT * FROM run_plan WHERE id=?", (execution["run_plan_id"],))
            plan = await cur.fetchone()
            await cur.execute(
                "SELECT * FROM branch_run WHERE execution_run_id=? ORDER BY started_at",
                (execution_id,),
            )
            branches = await cur.fetchall()
            branch_ids = [b["id"] for b in branches]
            nodes = []
            if branch_ids:
                placeholders = ",".join("?" * len(branch_ids))
                await cur.execute(
                    f"SELECT * FROM node_run WHERE branch_run_id IN ({placeholders}) ORDER BY started_at",
                    branch_ids,
                )
                nodes = await cur.fetchall()
            await cur.execute(
                "SELECT * FROM session_log WHERE execution_run_id=? ORDER BY created_at",
                (execution_id,),
            )
            session_logs = await cur.fetchall()
            return {
                "execution": dict(execution),
                "plan": dict(plan) if plan else {},
                "branches": [dict(b) for b in branches],
                "nodes": [dict(n) for n in nodes],
                "session_logs": [dict(log) for log in session_logs],
                "flow_name": plan["flow_name"] if plan else "未知",
            }

    def _build_summary(self, data: dict) -> str:
        exec_data = data.get("execution", {})
        plan_data = data.get("plan", {})
        elapsed = 0
        if exec_data.get("started_at") and exec_data.get("finished_at"):
            elapsed = int(exec_data["finished_at"] - exec_data["started_at"])
        items = [
            build_summary_item("流程名", plan_data.get("flow_name", "未知")),
            build_summary_item("状态", exec_data.get("status", "未知")),
            build_summary_item("分支数", str(len(data.get("branches", [])))),
            build_summary_item("节点数", str(len(data.get("nodes", [])))),
            build_summary_item("用时", f"{elapsed}s"),
            build_summary_item("发起人", plan_data.get("starter_username", "未知")),
        ]
        return "\n".join(items)

    def _build_branch_details(self, data: dict) -> str:
        branches = data.get("branches", [])
        if not branches:
            return "<div>无分支记录</div>"
        lines = []
        for branch in branches:
            lines.append(
                f'<div class="branch-header">'
                f'分支: {branch["branch_key"]} &#8212; {branch["status"]}'
                f'</div>'
            )
            for node in data.get("nodes", []):
                if node.get("branch_run_id") == branch["id"]:
                    duration = 0
                    if node.get("started_at") and node.get("finished_at"):
                        duration = int((node["finished_at"] - node["started_at"]) * 1000)
                    lines.append(build_node_item(node["node_name"], node["status"], duration))
        return "\n".join(lines)

    def _build_session_log(self, data: dict) -> str:
        logs = data.get("session_logs", [])
        if not logs:
            return '<div class="log-entry">无会话记录</div>'
        lines = []
        for log in logs:
            ts = time.strftime("%H:%M:%S", time.localtime(log.get("created_at", 0)))
            lines.append(build_log_entry(
                ts,
                log.get("username", ""),
                log.get("action", ""),
                log.get("reason", ""),
            ))
        return "\n".join(lines)
```
