```python
        # 获取当前步骤

        if step_widget and hasattr(step_widget, 'validate'):
```

```python
# tests/test_steps_integration.py
import pytest
from PySide6.QtWidgets import QApplication
from workflow_test_desktop.ui.main_window import MainWindow


@pytest.fixture
def app():
    app = QApplication.instance()
    if app is None:
        app = QApplication([])
    yield app


def test_step_account_validate(app):
    mw = MainWindow(config=None, secrets=None, db=None, loop=None)
    mw._show_step(0)
    step = mw._content_stack.layout().itemAt(0).widget()
    # 默认预填欧阳改，验证应通过
    if hasattr(step, 'validate'):
        valid, msg = step.validate()
        assert valid


def test_step_flow_validate_blocks_navigation(app):
    mw = MainWindow(config=None, secrets=None, db=None, loop=None)
    mw._show_step(1)
    step = mw._content_stack.layout().itemAt(0).widget()
    if hasattr(step, 'validate'):
        valid, msg = step.validate()
        assert not valid  # 未选流程时应失败
        assert "流程" in msg


def test_main_window_respects_validation(app):
    mw = MainWindow(config=None, secrets=None, db=None, loop=None)
    mw._show_step(1)
    initial_step = mw._current_step
    mw._on_next()  # 尝试跳过（应被拦截）
    assert mw._current_step == initial_step  # 未前进
```
